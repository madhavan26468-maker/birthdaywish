    <script>
    document.addEventListener('DOMContentLoaded', function() {

        // --- 1. Starfield Background Generation ---
        const starfield = document.getElementById('starfield-bg');
        const numStars = 150;

        for (let i = 0; i < numStars; i++) {
            const star = document.createElement('div');
            star.classList.add('star');
            const size = Math.random() * 3 + 1;
            star.style.width = `${size}px`;
            star.style.height = `${size}px`;
            star.style.top = `${Math.random() * 100}%`;
            star.style.left = `${Math.random() * 100}%`;
            star.style.animationDelay = `${Math.random() * 5}s`;
            star.style.animationDuration = `${Math.random() * 5 + 3}s`;
            starfield.appendChild(star);
        }

        // --- 2. Scroll-based Fade-in Animations ---
        const sections = document.querySelectorAll('.journey-section');

        const observerOptions = {
            root: null, // observes intersections relative to the viewport
            rootMargin: '0px',
            threshold: 0.2 // trigger when 20% of the element is visible
        };

        const sectionObserver = new IntersectionObserver((entries, observer) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('is-visible');
                    // Optional: unobserve after it has been animated
                    // observer.unobserve(entry.target);
                }
            });
        }, observerOptions);

        sections.forEach(section => {
            sectionObserver.observe(section);
        });


        // --- 3. Confetti Finale ---
        const confettiButton = document.getElementById('confetti-button');
        const finalMessage = document.getElementById('final-message');
        const canvas = document.getElementById('confetti-canvas');
        const ctx = canvas.getContext('2d');
        let confettiParticles = [];

        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        window.addEventListener('resize', () => {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        });

        const confettiColors = ['#ff69b4', '#ffda00', '#00eaff', '#ff0a54', '#f0f0f0'];

        function ConfettiParticle() {
            this.x = Math.random() * canvas.width;
            this.y = -20;
            this.size = Math.random() * 10 + 5;
            this.weight = Math.random() * 2 + 1;
            this.color = confettiColors[Math.floor(Math.random() * confettiColors.length)];
            this.rotation = Math.random() * 360;
            this.rotationSpeed = Math.random() * 2 - 1;
        }

        ConfettiParticle.prototype.update = function() {
            this.y += this.weight;
            this.x += Math.sin(this.y * 0.05) * 2;
            this.rotation += this.rotationSpeed;
            if (this.y > canvas.height + 20) {
                // Reset particle
                this.y = -20;
                this.x = Math.random() * canvas.width;
            }
        };

        ConfettiParticle.prototype.draw = function() {
            ctx.save();
            ctx.translate(this.x + this.size / 2, this.y + this.size / 2);
            ctx.rotate(this.rotation * Math.PI / 180);
            ctx.fillStyle = this.color;
            ctx.fillRect(-this.size / 2, -this.size / 2, this.size, this.size);
            ctx.restore();
        };

        function startConfetti() {
            confettiParticles = [];
            for (let i = 0; i < 200; i++) {
                confettiParticles.push(new ConfettiParticle());
            }
            animateConfetti();
        }
        
        let animationFrameId;
        function animateConfetti() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            confettiParticles.forEach(p => {
                p.update();
                p.draw();
            });
            animationFrameId = requestAnimationFrame(animateConfetti);
        }
        
        function stopConfetti() {
            cancelAnimationFrame(animationFrameId);
             // Smoothly fade out remaining confetti
            let fadeOut = setInterval(() => {
                ctx.fillStyle = 'rgba(26, 26, 26, 0.1)';
                ctx.fillRect(0, 0, canvas.width, canvas.height);
                if (ctx.globalAlpha <= 0.1) {
                    ctx.clearRect(0, 0, canvas.width, canvas.height);
                    clearInterval(fadeOut);
                }
            }, 50);
        }

        confettiButton.addEventListener('click', () => {
            confettiButton.style.display = 'none'; // Hide the button
            finalMessage.classList.add('visible'); // Show the final message
            
            startConfetti();

            // Stop the confetti after a few seconds
            setTimeout(() => {
                stopConfetti();
            }, 6000); // 6 seconds
        });

    });
    </script>
